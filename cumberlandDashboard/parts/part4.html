} function getFloodOpacity(level) { return 0.25 + (level / FLOOD_MAX) * 0.30; } // XYZ tile URL pattern — tiles pre-rendered by generate_tiles.py function floodTileUrl(side, levelKey) { return 'assets/tiles/' + side + '/' + levelKey + '/{z}/{x}/{y}.png'; } activeFundyLayer = floodTileCache[ckF]; map.addLayer(activeFundyLayer); activeFundyLayer.setVisible(showFlood); } if (currentSide === 'both' || currentSide === 'north') { var ckN = 'north_' + key; if (!floodTileCache[ckN]) { floodTileCache[ckN] = createFloodTileLayer('north', level); } activeNorthLayer = floodTileCache[ckN]; map.addLayer(activeNorthLayer); activeNorthLayer.setVisible(showFlood); } // Tiles load on-demand via HTTP — no heavy parsing needed // Small delay to let tile requests fire, then hide overlay setTimeout(function() { overlay.classList.remove('visible'); sliderTimeout = setTimeout(function() { updateFloodDisplay(level); }, 300); } sliderEl.addEventListener('input', function(e) { applyLevel(clampLevel(parseFloat(e.target.value))); }); inputEl.addEventListener('change', function(e) { applyLevel(clampLevel(parseFloat(e.target.value))); }); document.querySelectorAll('.side-btn').forEach(function(btn) { btn.addEventListener('click', function() { document.querySelectorAll('.side-btn').forEach(function(b) { b.classList.remove('active');
