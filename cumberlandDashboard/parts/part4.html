// ---------- FLOOD SLIDER LOGIC (COG TIFF EPSG:3857 — optimized) ---------- var FLOOD_MIN = 0; var FLOOD_MAX = 11; var FLOOD_STEP = 0.1; function levelToKey(level) { var deci = Math.round(level * 10); return Math.floor(deci / 10) + '_' + (deci % 10) + 'm'; } function getFloodOpacity(level) { return 0.25 + (level / FLOOD_MAX) * 0.30; } function cogTiffUrl(side, level) { return 'assets/cog_3857/' + (side === 'fundy' ? 'fundy' : 'north') + '/RasterFlood_' + levelToKey(level) + '.tif'; } // ABGR color for Uint32Array (little-endian: byte order is R,G,B,A in memory = ABGR as uint32) function makeFloodColor32(alpha) { return (alpha << 24) | (235 << 16) | (99 << 8) | 37; } // Parse COG → OL ImageStatic layer (EPSG:3857, no reprojection) // Uses Uint32Array for 4x faster pixel fill vs byte-by-byte async function loadCogTiffLayer(side, level) { var url = cogTiffUrl(side, level); var response = await fetch(url); if (!response.ok) throw new Error('HTTP ' + response.status); var arrayBuffer = await response.arrayBuffer(); var tiff = await GeoTIFF.fromArrayBuffer(arrayBuffer); var image = await tiff.getImage(); var width = image.getWidth(); var height = image.getHeight(); var origin = image.getOrigin(); var res = image.getResolution(); var rasterData = await image.readRasters({ samples: [0] }); var band = rasterData[0]; // Extent in EPSG:3857 var extent = [origin[0], origin[1] + height * res[1], origin[0] + width * res[0], origin[1]]; // Nodata var noData = null; if (image.fileDirectory.GDAL_NODATA !== undefined) { noData = parseFloat(image.fileDirectory.GDAL_NODATA); } // Uint32Array pixel fill — single 32-bit write per pixel instead of 4 byte writes var canvas = document.createElement('canvas'); canvas.width = width; canvas.height = height; var ctx = canvas.getContext('2d'); var imgData = ctx.createImageData(width, height); var buf32 = new Uint32Array(imgData.data.buffer); var floodColor = makeFloodColor32(Math.round(getFloodOpacity(level) * 255)); for (var i = 0, len = band.length; i < len; i++) { var v = band[i]; if (v !== 0 && v === v && v !== noData) { buf32[i] = floodColor; } } ctx.putImageData(imgData, 0, 0); // Create blob URL from canvas var blobUrl = await new Promise(function(resolve) { canvas.toBlob(function(blob) { resolve(blob ? URL.createObjectURL(blob) : canvas.toDataURL()); }, 'image/png'); }); return new ol.layer.Image({ source: new ol.source.ImageStatic({ url: blobUrl, imageExtent: extent, projection: 'EPSG:3857' }), zIndex: side === 'fundy' ? 10 : 11 }); } // Layer cache and active tracking var floodCache = {}; var activeFundyLayer = null; var activeNorthLayer = null; function removeActiveFloodLayers() { if (activeFundyLayer) { try { map.removeLayer(activeFundyLayer); } catch(e) {} activeFundyLayer = null; } if (activeNorthLayer) { try { map.removeLayer(activeNorthLayer); } catch(e) {} activeNorthLayer = null; } } async function updateFloodDisplay(level) { var overlay = document.getElementById('loading-overlay'); overlay.classList.add('visible'); removeActiveFloodLayers(); var displayVal = level.toFixed(1); document.getElementById('flood-level-display').textContent = displayVal; document.getElementById('kpi-flood-level').textContent = displayVal + 'm'; var showFlood = document.getElementById('layer-flood').checked; var key = levelToKey(level); var promises = []; if (currentSide === 'both' || currentSide === 'fundy') { var ckF = 'fundy_' + key; promises.push( (floodCache[ckF] ? Promise.resolve(floodCache[ckF]) : loadCogTiffLayer('fundy', level).then(function(l) { floodCache[ckF] = l; return l; })) .then(function(layer) { activeFundyLayer = layer; map.addLayer(layer); layer.setVisible(showFlood); }) .catch(function(e) { console.warn('Fundy:', e.message); }) ); } if (currentSide === 'both' || currentSide === 'north') { var ckN = 'north_' + key; promises.push( (floodCache[ckN] ? Promise.resolve(floodCache[ckN]) : loadCogTiffLayer('north', level).then(function(l) { floodCache[ckN] = l; return l; })) .then(function(layer) { activeNorthLayer = layer; map.addLayer(layer); layer.setVisible(showFlood); }) .catch(function(e) { console.warn('North:', e.message); }) ); } await Promise.all(promises); overlay.classList.remove('visible'); // Lazy prefetch: only preload the next level in the direction user was going, after a delay clearTimeout(window._prefetchTimer); window._prefetchTimer = setTimeout(function() { var next = Math.round((level + FLOOD_STEP) * 10) / 10; if (next <= FLOOD_MAX) { if ((currentSide === 'both' || currentSide === 'fundy') && !floodCache['fundy_' + levelToKey(next)]) { loadCogTiffLayer('fundy', next).then(function(l) { floodCache['fundy_' + levelToKey(next)] = l; }).catch(function(){}); } if ((currentSide === 'both' || currentSide === 'north') && !floodCache['north_' + levelToKey(next)]) { loadCogTiffLayer('north', next).then(function(l) { floodCache['north_' + levelToKey(next)] = l; }).catch(function(){}); } } }, 2000); } function clampLevel(val) { if (isNaN(val)) return currentFloodLevel; return Math.max(FLOOD_MIN, Math.min(FLOOD_MAX, Math.round(val * 10) / 10)); } var sliderTimeout = null; var sliderEl = document.getElementById('flood-slider'); var inputEl = document.getElementById('flood-level-input'); var btnUp = document.getElementById('level-up'); var btnDown = document.getElementById('level-down'); function applyLevel(level) { currentFloodLevel = level; var d = level.toFixed(1); sliderEl.value = d; inputEl.value = d; document.getElementById('flood-level-display').textContent = d; document.getElementById('kpi-flood-level').textContent = d + 'm'; if (sliderTimeout) clearTimeout(sliderTimeout); sliderTimeout = setTimeout(function() { updateFloodDisplay(level); }, 300); } sliderEl.addEventListener('input', function(e) { applyLevel(clampLevel(parseFloat(e.target.value))); }); inputEl.addEventListener('change', function(e) { applyLevel(clampLevel(parseFloat(e.target.value))); }); btnUp.addEventListener('click', function() { applyLevel(clampLevel(currentFloodLevel + FLOOD_STEP)); }); btnDown.addEventListener('click', function() { applyLevel(clampLevel(currentFloodLevel - FLOOD_STEP)); }); document.querySelectorAll('.side-btn').forEach(function(btn) { btn.addEventListener('click', function() { document.querySelectorAll('.side-btn').forEach(function(b) { b.classList.remove('active'); }); this.classList.add('active'); currentSide = this.getAttribute('data-side'); updateFloodDisplay(currentFloodLevel); }); }); // ---------- ROAD NETWORK ---------- fetch('assets/geojson/roads_cumberland.geojson') .then(function(r) { return r.json(); }) .then(function(data) { var fmt = new ol.format.GeoJSON(); var features = fmt.readFeatures(data, { featureProjection: 'EPSG:3857' }); roadsLayer.setStyle(function(feature) { var hw = feature.get('highway'); var color = '#f39c12', width = 2; if (hw === 'motorway' || hw === 'motorway_link') { color = '#8e44ad'; width = 4; } else if (hw === 'trunk' || hw === 'trunk_link') { color = '#c0392b'; width = 3.5; } else if (hw === 'primary') { color = '#e67e22'; width = 3; } else if (hw === 'secondary') { color = '#f39c12'; width = 2.5; } return new ol.style.Style({ stroke: new ol.style.Stroke({ color: color, width: width }) }); }); roadsLayer.getSource().addFeatures(features); showToast('Road network loaded (' + features.length + ' segments)', 'info'); }) .catch(function(err) { console.error('Road network error:', err); }); // ---------- CRITICAL INFRASTRUCTURE ---------- function getInfraStyle(amenity) { if (amenity === 'fire_station') return { color: '#c0392b' }; if (amenity === 'hospital' || amenity === 'clinic') return { color: '#27ae60' }; if (amenity === 'police') return { color: '#2980b9' }; if (amenity === 'townhall') return { color: '#8e44ad' }; if (amenity === 'community_centre') return { color: '#e67e22' }; return { color: '#7f8c8d' }; } fetch('assets/geojson/infrastructure_cumberland.geojson') .then(function(r) { return r.json(); }) .then(function(data) { var fmt = new ol.format.GeoJSON(); var features = fmt.readFeatures(data, { featureProjection: 'EPSG:3857' }); var count = 0; features.forEach(function(f) { var amenity = f.get('amenity') || f.get('healthcare') || f.get('emergency') || ''; var s = getInfraStyle(amenity); f.setStyle(new ol.style.Style({ image: new ol.style.Circle({ radius: 6, fill: new ol.style.Fill({ color: s.color }), stroke: new ol.style.Stroke({ color: '#fff', width: 2 }) }) })); count++; }); infraLayer.getSource().addFeatures(features); document.getElementById('kpi-infra').textContent = count; showToast('Infrastructure loaded (' + count + ' facilities)', 'info'); }) .catch(function(err) { console.error('Infrastructure error:', err); });
